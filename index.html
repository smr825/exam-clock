        // 시간 데이터를 초 단위로 변환하는 함수 (비교용)
        function timeToSeconds(timeStr) {
            const [h, m] = timeStr.split(':');
            return parseInt(h) * 3600 + parseInt(m) * 60;
        }
        // 시험 일정 데이터 정의
        const scheduleData = [
            { id: "p1", name: "1교시 준비", start: timeToSeconds("08:45"), end: 
timeToSeconds("08:50"), isEval: false },
            { id: "p1", name: "1교시 평가", start: timeToSeconds("08:50"), end: 
timeToSeconds("09:40"), isEval: true },
            { id: null, name: "쉬는 시간", start: timeToSeconds("09:40"), end: 
timeToSeconds("09:55"), isEval: false },
            { id: "p2", name: "2교시 자율학습", start: timeToSeconds("09:55"), end: 
timeToSeconds("10:50"), isEval: true },
            { id: null, name: "쉬는 시간", start: timeToSeconds("10:50"), end: 
timeToSeconds("11:05"), isEval: false },
            { id: "p3", name: "3교시 준비", start: timeToSeconds("11:05"), end: 
timeToSeconds("11:10"), isEval: false },
            { id: "p3", name: "3교시 평가", start: timeToSeconds("11:10"), end: 
timeToSeconds("12:00"), isEval: true }
        ];
        let audioCtx = null;
        let previousStateName = null;
        // 웹 오디오 API를 이용한 알림음 생성 (외부 파일 불필요)
        function initAudio() {
            if (!audioCtx) {
                audioCtx = new (window.AudioContext || 
window.webkitAudioContext)();
                document.getElementById('audio-init-overlay').style.display = 'none';
            }
        }
        function playBeep() {
            if (!audioCtx) return;
            if (audioCtx.state === 'suspended') audioCtx.resume();
            
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            
            osc.type = 'sine';
            osc.frequency.setValueAtTime(880, audioCtx.currentTime); // 880Hz (A5
음)
            
            gain.gain.setValueAtTime(0.5, audioCtx.currentTime);
            gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 
1.5);
            
            osc.connect(gain);
            gain.connect(audioCtx.destination);
            
            osc.start();
            osc.stop(audioCtx.currentTime + 1.5);
        }
        // 시계 및 일정 업데이트 메인 함수
        function updateClock() {
            const now = new Date();
            const hours = String(now.getHours()).padStart(2, '0');
            const minutes = String(now.getMinutes()).padStart(2, '0');
            const seconds = String(now.getSeconds()).padStart(2, '0');
            
            // 시계 텍스트 업데이트
            document.getElementById('time').innerText = 
`${hours}:${minutes}:${seconds}`;
            // 날짜 업데이트
            const days = ['일', '월', '화', '수', '목', '금', '토'];
            const year = now.getFullYear();
            const month = String(now.getMonth() + 1).padStart(2, '0');
            const date = String(now.getDate()).padStart(2, '0');
            const day = days[now.getDay()];
            document.getElementById('date').innerText = `${year}년 ${month}월 
${date}일 ${day}요일`;
            // 현재 상태 계산
            const currentSeconds = now.getHours() * 3600 + now.getMinutes() * 60 
+ now.getSeconds();
            let currentState = null;
            for (let i = 0; i < scheduleData.length; i++) {
                if (currentSeconds >= scheduleData[i].start && currentSeconds < 
scheduleData[i].end) {
                    currentState = scheduleData[i];
                    break;
                }
            }
            // 모든 하이라이트 초기화
            document.querySelectorAll('.schedule-item').forEach(el => 
el.classList.remove('active'));
            if (currentState) {
                // 현재 진행중인 교시 배경 하이라이트
                if (currentState.id) {
                    document.getElementById(currentState.id).classList.add('active');
                }
                // 남은 시간 계산 (분:초)
                const remainSec = currentState.end - currentSeconds;
                const rMin = String(Math.floor(remainSec / 60)).padStart(2, '0');
                const rSec = String(remainSec % 60).padStart(2, '0');
                document.getElementById('current-period-name').innerText = 
`${currentState.name} 진행 중`;
                document.getElementById('remaining-time').innerText = `남은 시간 
${rMin}:${rSec}`;
                document.getElementById('remaining-time').style.color = 
currentState.isEval ? '#ef4444' : '#fbbf24';
                // 교시가 바뀌었을 때 알림음 재생
                if (previousStateName !== currentState.name) {
                    if (previousStateName !== null) playBeep();
                    previousStateName = currentState.name;
                }
            } else {
                if (currentSeconds < scheduleData[0].start) {
                    document.getElementById('current-period-name').innerText = "시
험 대기 중";
                } else {
                    document.getElementById('current-period-name').innerText = "모
든 일정 종료";
                }
                document.getElementById('remaining-time').innerText = "--:--";
                previousStateName = null;
            }
        }
        // 1초마다 업데이트 실행
        setInterval(updateClock, 1000);
        updateClock(); // 초기 즉시 실행
